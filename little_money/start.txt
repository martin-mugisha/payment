from config.models import Merchant
from config.utils import generate_api_key  # if defined elsewhere, or paste the function here
import secrets, string

def generate_api_key(length=40):
    alphabet = string.ascii_letters + string.digits
    return ''.join(secrets.choice(alphabet) for _ in range(length))

merchant = Merchant.objects.create(
    name="Demo Merchant",
    api_key=generate_api_key(),
    api_secret=generate_api_key(),
    allowed_modes=["sandbox"],
    default_mode="sandbox",
    is_active=True
)

print("API KEY:", merchant.api_key)
print("API SECRET:", merchant.api_secret)
